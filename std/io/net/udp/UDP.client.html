    Function UDP.client

    Connect to resolved hostname and port. Uses the first resolved address the socket can successfully connect to.

    UDP client(S) (
      S hostname,
      ushort port
    )
    if (isStringLike!S);

    UDP is a connection-less protocol. Connecting to a client to a remote address allows to use send without specifying a destination address. It will also apply a filter to only receive traffic from the connected address.

    Parameters

    NameDescription
    hostname remote hostname to connect to
    port remote port to connect to

    Example

    auto server = UDP.server("localhost", 1234);
// connect to remote hostname and port
auto client = UDP.client("localhost", 1234);

// send to connected address
ubyte[4] ping = ['p', 'i', 'n', 'g'];
client.send(ping[]);
ubyte[4] buf;
assert(server.recv(buf[]) == 4);
assert(buf[] == ping[]);

    Function UDP.client

    Connect to remote socket address.

    UDP client(SocketAddr) (
      auto in ref SocketAddr addr
    ) @trusted
    if (isSocketAddr!SocketAddr);

    Parameters

    NameDescription
    addr remote socket address to connect to

    Example

    auto server = UDP.server("localhost");
// "connect" to remote address
auto client = UDP.client(server.localAddr);

// send to connected address
ubyte[4] ping = ['p', 'i', 'n', 'g'];
client.send(ping[]);
ubyte[4] buf;
assert(server.recv(buf[]) == 4);
assert(buf[] == ping[]);

// can still use sendTo with other addr (unless you're on BSD/OSX)
version (OSX) {}
else
{
    auto server2 = UDP.server("localhost");
    client.sendTo(server2.localAddr, ping[]);
    assert(server2.recv(buf[]) == 4);
    assert(buf[] == ping[]);
}

// while keep sending to connected address
client.send(ping[]);
assert(server.recv(buf[]) == 4);
assert(buf[] == ping[]);

// client can only receive data from connected server
server.sendTo(client.localAddr, ping[]);
assert(client.recv(buf[]) == 4);
assert(buf[] == ping[]);
